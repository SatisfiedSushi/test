local Tool = script.Parent
local humanoid = nil
local eater = nil
local con = nil
local debounce = false
local player = nil
local config = Tool.Configuration
local poison = math.random(1, 10)
local uses = math.random(config.MinUses.Value, config.MaxUses.Value)
print(uses)
poisonedValue = 1

function onEquipped(mouse)
	player = game.Players:GetPlayerFromCharacter(Tool.Parent)
	humanoid = Tool.Parent:FindFirstChild("Humanoid")
	if humanoid ~= nil then
		eat = humanoid:LoadAnimation(Tool.Eat)
	end
	con = mouse.Button1Down:connect(function() onButton1Down(mouse) end)
end

function onUnequipped(mouse)
	humanoid = nil
	if eat ~= nil then
		eat:remove()
		eat = nil
	end
	if con ~= nil then
		con:disconnect()
	end
end
function poisonPlayer()
	s = player.Character:FindFirstChild("PoisonScript")
	if s ~= nil then
		s.Poison.Value = s.Poison.Value + 1
	else
		s = script.PoisonScript:clone()
		s.Parent = player.Character
		s.Disabled = false
	end
end


function onButton1Down(mouse)
	if debounce == false then
		debounce = true
		eat:Play()
		uses = uses - 1
		if player.PlayerGui:FindFirstChild("HungerGUI") then
			local hunger = math.random(config.MinHungerAmount.Value, config.MaxHungerAmount.Value)
			player.PlayerGui.HungerGUI.Configuration.Hunger.Value = player.PlayerGui.HungerGUI.Configuration.Hunger.Value + hunger
		end
		if player.PlayerGui:FindFirstChild("ThirstGUI") then
			local thirst = math.random(config.MinThirstAmount.Value, config.MaxThirstAmount.Value)
			player.PlayerGui.ThirstGUI.Configuration.Thirst.Value = player.PlayerGui.ThirstGUI.Configuration.Thirst.Value + thirst
		end
		local pitch = math.random(850, 1125)
		if string.len(tostring(pitch)) == 3 then
			pitch = tonumber("0." .. tostring(pitch))
		else
			pitch = tonumber("1." .. string.sub(tostring(pitch), 2, 4))
		end
		if poison <= config.ChanceToBePoison.Value then
			poisonedValue = poisonedValue + 1
			poisonPlayer()
		end
		Tool.Handle.EatSound.Pitch = pitch
		Tool.Handle.EatSound:Play()
		wait(2)
		debounce = false
		if uses == 0 then
			Tool:remove()
		end
	end
end

Tool.Equipped:connect(onEquipped)
Tool.Unequipped:connect(onUnequipped)
